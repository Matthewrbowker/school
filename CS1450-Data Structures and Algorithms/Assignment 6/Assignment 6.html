 31     public void addCar(String tag) {
 32         // Method to add the cars to the proper places
 33         // string input, console output.  
 34         
 35         car tempCar = new car(tag);    // Temporary car - added to wherever
 36         
 37         if (alley.size() < 6) {
 38             // If our alley isn't full...
 39             
 40             tempCar.setActionCode('A');  //Set the action code of the car
 41             alley.insert(tempCar);       // Add the car to the alley
 42             
 43             // Print some user output
 44             System.out.println("Car #" + tag + " added to the alley.");
 45         }
 46         else if (driveway.size() < 3) {
 47             // If our alley is full but the driveway isn't...
 48             
 49             tempCar.setActionCode('B');  // Set the action code of the car
 50             driveway.push(tempCar);      // Add the car to the driveway
 51             
 52             // Print some user output
 53             System.out.println("Car #" + tag + " added to the driveway.");
 54         }
 55         else {
 56             // We're all full here... turning the car away
 57             
 58             numTurnedAway++;  // Add one to the number of cars turned away
 59             
 60             // User output
 61             System.out.println("Car #" + tag + " turned away.");
 62             System.out.println("Now removing " + (numTurnedAway+1) + " cars.");
 63         }

 67     public void removeCar() {
 68         // Method to systematically remove the cars that come in
 69         // No input, console output
 70         
 71         int count;                  // Counter variable for the loop
 72         car tempCar;                // temprary car, it's what was removed
 73         String buffer = "Removed:"; // Buffer for display to the user
 74         boolean display = false;    // Should we display the buffer?
 75         
 76         for (count = 0; count < 1 + numTurnedAway; count++) {
 77             // Loop looks through, removes cars from the alley first
 78             // until we agree with eddie's demands
 79             // No input, console output
 80             
 81             if (!alley.isEmpty()) {
 82                 // Remove cars from the alley first
 83                 
 84                 // Get a car out
 85                 tempCar = alley.remove();
 86                 
 87                 // Add the tag to the string buffer
 88                 buffer = buffer + " #" + tempCar.getTag();
 89                 
 90                 // Hey, we did something!  We need to show the buffer.
 91                 display = true;
 92             }
 93             else if (!driveway.isEmpty()) {
 94                 // Now removing cars from the driveway
 95                 
 96                 // Get the car out
 97                 tempCar = driveway.pop();
 98                 
 99                 // Add the tag to the string buffer
100                 buffer = buffer + " #" + tempCar.getTag();
101                 
102                 // We did something, display the buffer.
103                 display = true;
104             }
105             else {
106                 // Do nothing ...  there's nothing to remove!
107             }
108         }
109         
110         if (alley.isEmpty() && driveway.isEmpty()) {
111             // If we emptied both the alley and the driveway, Eddie calms
112             // down a little bit.  So, let's remove one less car next time.
113             
114             numTurnedAway--; // Decriment the number
115             
116             // Display the new number to the user
117             System.out.println("Now removing " + (numTurnedAway+1) + " cars.");
118         }
119         
120         if (display) {
121             // If we did something, display the text buffer
122             System.out.println(buffer);
123         }
124         
125         // Print out our new stack and queue.
126         displayCars();
127     }
