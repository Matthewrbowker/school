11 public class BST {
12 
13     public static void main(String[] args) {
14         BSTree tree;           // The Binary Search Tree we'll be using
15         TextReader reader;     // The TextReader class, provided by Al.
16         String nextWord;       // The next word we're adding to the BST
17         
18         // Intitialize the tree and the text reader.
19         tree = new BSTree();
20         reader = new TextReader("/Users/matthewrbowker/"
21                 + "Documents/School/DSA/Assignments/Assignment 7/Data.as7");
22         
23         // Read the first word of the file
24         nextWord = reader.GetWord();
25         
26         // Loop through the file, reading the other words.
27         while (nextWord != null) {
28             tree.insert(nextWord, true);
29             nextWord = reader.GetWord();
30         }
31         
32         // Display the tree.
33         tree.Display();
34     }
35 }
